39 namespace sci {
40 class Hash {
41 public:
42  SHA256_CTX hash;
43  char buffer[HASH_BUFFER_SIZE];
44  int size = 0;
45  static const int DIGEST_SIZE = 32;
46  Hash() { SHA256_Init(&hash); }
47  ~Hash() {}
48  void put(const void *data, int nbyte) {
49  if (nbyte > HASH_BUFFER_SIZE)
50  SHA256_Update(&hash, data, nbyte);
51  else if (size + nbyte < HASH_BUFFER_SIZE) {
52  memcpy(buffer + size, data, nbyte);
53  size += nbyte;
54  } else {
55  SHA256_Update(&hash, (char *)buffer, size);
56  memcpy(buffer, data, nbyte);
57  size = nbyte;
58  }
59  }
60  void put_block(const block128 *block128, int nblock = 1) {
61  put(block128, sizeof(block128) * nblock);
62  }
63  void digest(char *a) {
64  if (size > 0) {
65  SHA256_Update(&hash, (char *)buffer, size);
66  size = 0;
67  }
68  SHA256_Final((unsigned char *)a, &hash);
69  }
70  void reset() {
71  SHA256_Init(&hash);
72  size = 0;
73  }
74  static void hash_once(void *digest, const void *data, int nbyte) {
75  (void)SHA256((const unsigned char *)data, nbyte, (unsigned char *)digest);
76  }
77  __attribute__((target("sse2"))) static block128
78  hash_for_block128(const void *data, int nbyte) {
79  // even though stack is aligned to 16 byte, we don't know the order of
80  // locals.
81  alignas(16) char digest[DIGEST_SIZE];
82  hash_once(digest, data, nbyte);
83  return _mm_load_si128((__m128i *)&digest[0]);
84  }
85 
86  __attribute__((target("avx"))) static block256
87  hash_for_block256(const void *data, int nbyte) {
88  alignas(32) char digest[DIGEST_SIZE];
89  hash_once(digest, data, nbyte);
90  return _mm256_load_si256((__m256i *)&digest[0]);
91  }

93  static block128 KDF128(emp::Point &in, uint64_t id = 1) {
94  size_t len = in.size();
95  in.group->resize_scratch(len + 8);
96  unsigned char *tmp = in.group->scratch;
97  in.to_bin(tmp, len);
98  memcpy(tmp + len, &id, 8);
99  block128 ret = hash_for_block128(tmp, len + 8);
