377 /*
378  Given a byte array arr of length arrLen, interpret this as a bit array,
379  and read from bitIdx = bitIdx, #bits = bitsToRead.
380  Return the read value as uint64_t (assuming obviously that bitsToRead <= 64).
381  The answer is stored in the lower bitsToRead bits of the returned value (lsb onwards).
382  Note: In the new implementation, while reading, this reads a uint64_t. Ensure the array
383  from which reading is being done has one uint64_t extra at the end to prevent reading
384  out of bounds.
385  This code assumes little endian encoding for the system
386 */
387 static uint64_t readFromPackedArr(uint8_t* arr, int arrLen, uint64_t bitIdx, uint64_t bitsToRead){
388  assert(bitsToRead<=64);
389  uint64_t firstByteIdx = bitIdx/8;
390  uint64_t toRejectBitsInFirstByte = bitIdx%8;
391  uint64_t ans = *((uint64_t*)(arr+firstByteIdx));
392  ans >>= toRejectBitsInFirstByte;
393  if (bitsToRead > 64 - toRejectBitsInFirstByte){
394  // Read from the unaligned byte at firstByteIdx+8
395  ans = ans & ((1ULL<<(64 - toRejectBitsInFirstByte))-1);
396  uint64_t temp = ((uint64_t)(arr[firstByteIdx+8])) & ((1ULL<<toRejectBitsInFirstByte)-1);
397  ans += (temp<<(64 - toRejectBitsInFirstByte));
398  }
399  ans = ans & all1Mask(bitsToRead);
400  return ans;
401 }

411 /*
412  Similar to above, given a byte array arr of length = arrLen, interpret this as a bit array,
413  and start writing bits at bitIdx = bitIdx with #bits = bitlen. Actual bits to be written are given
414  in the lower bits of val, i.e. lower bitlen bits of val represent the bits to be written.
415  Obviously again bitlen <= 64.
416  Note that in the final unaligned byte this writes 0 in the extra positions.
417  Also, assuming extra bits of val after bitlen are filled with 0s.
418  Note: The new implementation assumes that there one extra uint64_t space at the end.
419  This is because it writes the whole uint64_t given and so to prevent writing out of bounds,
420  ensure the array has one extra uint64_t space at the end.
421  Note: this code assumes little endian encoding
422 */
423 static void writeToPackedArr(uint8_t* arr, int arrLen, uint64_t bitIdx, uint64_t bitlen, uint64_t val){
424  assert(bitlen<=64);
425  //First write the unaligned bits -- i.e. first byte which needs to be written
426  uint64_t firstByteIdx = bitIdx/8;
427  uint64_t freeBitsInFirstByte = 8 - (bitIdx%8);
428  uint8_t valLowerByte = val; //This will give the lower 8 bits of val
429  writeInAByte(arr+firstByteIdx,bitIdx%8,valLowerByte); //If bitlen < freeBitsInFirstByte, this will write out 0 in those extra positions
430  // Now things are byte aligned
431  uint64_t valIter = (val >> freeBitsInFirstByte); //Since val is unsigned, this is unsigned right shift
432  (*((uint64_t*)(arr+firstByteIdx+1))) = valIter;
433 }

440 inline int64_t signed_val(uint64_t x, int bw_x) {
441  uint64_t pow_x = (bw_x == 64? 0ULL: (1ULL << bw_x));
442  uint64_t mask_x = pow_x - 1;
443  x = x & mask_x;
444  return int64_t(x - ((x >= (pow_x/2)) * pow_x));
