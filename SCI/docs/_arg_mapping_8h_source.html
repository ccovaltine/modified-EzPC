29 inline bool parseArgs(int argc, char *argv[], argmap_t &argmap) {
30  for (long i = 1; i < argc; i++) {
31  char *x = argv[i];
32  long j = 0;
33  while (x[j] != '=' && x[j] != '\0')
34  j++;
35  if (x[j] == '\0')
36  return false;
37  std::string arg(x, j);
38  if (argmap[arg] == NULL)
39  return false;
40  argmap[arg] = x + j + 1;
41  }
42  return true;
43 }

73 class ArgMapping {

121 void ArgMapping::parse(int argc, char **argv) {
122  for (long i = 1; i < argc; i++) {
123  const char *x = argv[i];
124  long j = 0;
125  while (x[j] != '=' && x[j] != '\0')
126  j++;
127  if (x[j] == '\0')
128  usage(argv[0]);
129  std::string name(x, j);
130  const char *s = x + j + 1;
131 
132  std::shared_ptr<ArgProcessor> ap = map[name];
133  if (!ap)
134  return usage(argv[0]);
135  if (!ap->process(s))
136  usage(argv[0]);
137  }
138 }
