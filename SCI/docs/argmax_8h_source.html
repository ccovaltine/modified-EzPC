46  // Constructor
47  ArgMaxProtocol(int party, int algeb_str, sci::IOPack *iopack, int l, int b,
48  uint64_t prime, sci::OTPack *otpack) {
49  this->party = party;
50  this->algeb_str = algeb_str;
51  this->iopack = iopack;
52  this->l = l;
53  mask_lower = (1ULL << this->l) - 1;
54  mask_upper = (1ULL << (2 * this->l)) - 1 - mask_lower;
55  this->b = b;
56  this->prime_mod = prime;
57  this->otpack = otpack;
58  if (algeb_str == RING) {
59  this->relu_oracle =
60  new ReLURingProtocol<type>(party, RING, iopack, l, b, otpack);
61  } else {
62  this->relu_field_oracle = new ReLUFieldProtocol<type>(
63  party, FIELD, iopack, l, b, this->prime_mod, otpack);
64  }
65  configure();
66  }

78  mask_l = (type)((1ULL << l) - 1);

80  mask_l = -1;
81  } else { // l = 64
82  mask_l = -1ULL;

86  int next_eight_multiple(int val) {
87  for (int i = 0; i < 8; i++) {
88  if ((val + i) % 8 == 0) {
89  return (val + i);
90  }
91  }
92  return 0;

95  void ArgMaxMPC(int size, type *inpArr, type *maxi, bool get_max_too = false,
96  type *max_val = nullptr) {
97  type *input_temp = new type[size + 16];
98  type *input_argmax_temp = new type[size + 16];
99  for (int i = 0; i < size; i++) {
100  input_temp[i] = inpArr[i];
101  input_argmax_temp[i] = 0;
102  }
103  if (party == sci::ALICE) {
104  for (type i = 0; i < (type)size; i++) {
105  input_argmax_temp[i] = i;
106  }
107  }
108  if (size & 1) {
109  input_temp[size] = input_temp[size - 1];
110  input_argmax_temp[size] = input_argmax_temp[size - 1];
111  size += 1;
112  }
113  type *compare_with = new type[size + 16];
114  type *compare_with_argmax = new type[size + 16];
115  type *relu_res = new type[size + 16];
116  type *argmax_res = new type[size + 16];
117  int no_of_nodes = size;
118  int no_of_nodes_child;
119  int pad1, pad2;
120  int times_stuck_on_8 = 0;
121 
122  while (no_of_nodes > 1) {
123  // std::cout<<"#nodes = "<<no_of_nodes<<std::endl;
124  no_of_nodes_child = (int)ceil((float)no_of_nodes / (float)2);
125  pad1 = next_eight_multiple(no_of_nodes_child);
126  pad1 = pad1 - no_of_nodes_child;
127  pad2 = no_of_nodes + 2 * pad1;
128  no_of_nodes_child += pad1;
129  if (no_of_nodes_child == 8) {
130  times_stuck_on_8++;
131  }
132  if (times_stuck_on_8 >= 5) {
133  // The backend code only supports a minimum batch size of 8
134  // So, whenever we have less than 8 child nodes, we pad it to get 8
135  // nodes. The child nodes are = 8 in the following cases: #parentnodes =
136  // 16, 8, 4, 2, 1. We will get the argmax result when parent nodes = 1.
137  // So, times_stuck_on_8 >= 5.
138  break;
139  }
140  for (int i = no_of_nodes; i < pad2; i++) {
141  input_temp[i] = input_temp[no_of_nodes - 1];
142  input_argmax_temp[i] = input_argmax_temp[no_of_nodes - 1];
143  }
144  if (this->algeb_str == FIELD) {
145  for (int i = 0; i < (pad2); i += 2) {
146  compare_with[i / 2] = sci::neg_mod(
147  (int64_t)(input_temp[i] - input_temp[i + 1]), this->prime_mod);
148  compare_with_argmax[i / 2] = sci::neg_mod(
149  (int64_t)(input_argmax_temp[i] - input_argmax_temp[i + 1]),
150  this->prime_mod);
151  }
152  } else { // RING
153  for (int i = 0; i < (pad2); i += 2) {
154  compare_with[i / 2] = (input_temp[i] - input_temp[i + 1]);
155  compare_with_argmax[i / 2] =
156  (input_argmax_temp[i] - input_argmax_temp[i + 1]);
157  }
158  }
159  if (this->l > 32) {
160  argmax_this_level_super_32(argmax_res, relu_res, compare_with_argmax,
161  compare_with, no_of_nodes_child);
162  } else {
163  argmax_this_level_sub_32(argmax_res, relu_res, compare_with_argmax,
164  compare_with, no_of_nodes_child);
165  }
166  if (this->algeb_str == FIELD) {
167  for (int i = 0; i < (no_of_nodes_child); i++) {
168  input_temp[i] =
169  (relu_res[i] + input_temp[2 * i + 1]) % this->prime_mod;
170  input_argmax_temp[i] =
171  (argmax_res[i] + input_argmax_temp[2 * i + 1]) % this->prime_mod;
172  }
173  } else { // RING
174  for (int i = 0; i < (no_of_nodes_child); i++) {
175  input_temp[i] = (relu_res[i] + input_temp[2 * i + 1]) & mask_l;
176  input_argmax_temp[i] =
177  (argmax_res[i] + input_argmax_temp[2 * i + 1]) & mask_l;
178  }
179  }
180  no_of_nodes = no_of_nodes_child;
181  }
182  maxi[0] = input_argmax_temp[0];
183  if (get_max_too) {
184  max_val[0] = input_temp[0];
185  }
186  if (this->algeb_str == RING) {
187  maxi[0] &= mask_l;
188  if (get_max_too) {
189  max_val[0] &= mask_l;
190  }
191  }
192 
193  delete[] argmax_res;
194  delete[] relu_res;
195  delete[] compare_with_argmax;
196  delete[] compare_with;
197  delete[] input_temp;
198  delete[] input_argmax_temp;
199  }

205  void argmax_this_level_super_32(type *argmax, type *result, type *indexshare,
206  type *share, int num_relu) {
207  uint8_t *drelu_ans = new uint8_t[num_relu];
208  if (this->algeb_str == FIELD) {
209  relu_field_oracle->relu(result, share, num_relu, drelu_ans, true);
210  } else { // RING
211  relu_oracle->relu(result, share, num_relu, drelu_ans, true);
212  }
213 
214  // Now perform x.msb(x)
215  // 2 OTs required with reversed roles
216  sci::block128 *ot_messages_0 = new sci::block128[num_relu];
217  sci::block128 *ot_messages_1 = new sci::block128[num_relu];
218 
219  uint64_t *additive_masks = new uint64_t[num_relu * 2];
220  sci::block128 *received_shares = new sci::block128[num_relu];
221  uint64_t *received_shares_0 = new uint64_t[num_relu];
222  uint64_t *received_shares_1 = new uint64_t[num_relu];
223 
224  if (this->algeb_str == FIELD) {
225  this->relu_field_oracle->triple_gen->prg->template random_mod_p<type>(
226  (type *)additive_masks, 2 * num_relu, this->prime_mod);
227  for (int i = 0; i < 2 * num_relu; i++) {
228  additive_masks[i] %= this->prime_mod;
229  }
230  } else { // RING
231  this->relu_oracle->triple_gen->prg->random_data(
232  additive_masks, 2 * num_relu * sizeof(type));
233  }
234  for (int i = 0; i < num_relu; i++) {
235  set_argmax_end_ot_messages_super_32(
236  ot_messages_0 + i, ot_messages_1 + i, share + i, indexshare + i,
237  drelu_ans + i, ((type *)additive_masks) + i, num_relu);
238  }
239 #pragma omp parallel num_threads(2)
240  {
241  if (omp_get_thread_num() == 1) {
242  if (party == sci::ALICE) {
243  if (this->algeb_str == FIELD) {
244  relu_field_oracle->otpack->iknp_reversed->recv(
245  received_shares, (bool *)drelu_ans, num_relu);
246  } else {
247  relu_oracle->otpack->iknp_reversed->recv(
248  received_shares, (bool *)drelu_ans, num_relu);
249  }
250  } else { // party == sci::BOB
251  if (this->algeb_str == FIELD) {
252  relu_field_oracle->otpack->iknp_reversed->send(
253  ot_messages_0, ot_messages_1, num_relu);
254  } else {
255  relu_oracle->otpack->iknp_reversed->send(ot_messages_0,
256  ot_messages_1, num_relu);
257  }
258  }
259  } else {
260  if (party == sci::ALICE) {
261  if (this->algeb_str == FIELD) {
262  relu_field_oracle->otpack->iknp_straight->send(
263  ot_messages_0, ot_messages_1, num_relu);
264  } else {
265  relu_oracle->otpack->iknp_straight->send(ot_messages_0,
266  ot_messages_1, num_relu);
267  }
268  } else { // party == sci::BOB
269  if (this->algeb_str == FIELD) {
270  relu_field_oracle->otpack->iknp_straight->recv(
271  received_shares, (bool *)drelu_ans, num_relu);
272  } else {
273  relu_oracle->otpack->iknp_straight->recv(
274  received_shares, (bool *)drelu_ans, num_relu);
275  }
276  }
277  }
278  }
279  for (int i = 0; i < num_relu; i++) {
280  received_shares_0[i] = _mm_extract_epi64(received_shares[i], 1);
281  received_shares_1[i] = _mm_extract_epi64(received_shares[i], 0);
282  }
283  for (int i = 0; i < num_relu; i++) {
284  result[i] = ((type *)additive_masks)[i] +
285  ((type *)received_shares_0)[(8 / sizeof(type)) * i];
286  argmax[i] = ((type *)additive_masks)[i + num_relu] +
287  ((type *)received_shares_1)[(8 / sizeof(type)) * i];
288  if (this->algeb_str == FIELD) {
289  result[i] %= this->prime_mod;
290  argmax[i] %= this->prime_mod;
291  }
292  }
293 
294  delete[] additive_masks;
295  delete[] received_shares;
296  delete[] received_shares_0;
297  delete[] received_shares_1;
298  delete[] ot_messages_0;
299  delete[] ot_messages_1;
300  delete[] drelu_ans;
301  }

303  void set_argmax_end_ot_messages_super_32(sci::block128 *ot_messages_0,
304  sci::block128 *ot_messages_1,
305  type *value_share, type *index_share,
306  uint8_t *xor_share,
307  type *additive_mask, int num_relu) {
308  type temp0, temp1, temp2, temp3;
309  if (this->algeb_str == FIELD) {
310  temp0 = sci::neg_mod((int64_t)value_share[0] - (int64_t)additive_mask[0],
311  this->prime_mod);
312  temp1 = sci::neg_mod((int64_t)0LL - (int64_t)additive_mask[0],
313  this->prime_mod);
314  temp2 = sci::neg_mod((int64_t)index_share[0] -
315  (int64_t)additive_mask[0 + num_relu],
316  this->prime_mod);
317  temp3 = sci::neg_mod((int64_t)0LL - (int64_t)additive_mask[0 + num_relu],
318  this->prime_mod);
319  } else { // RING
320  temp0 = (value_share[0] - additive_mask[0]);
321  temp1 = (0 - additive_mask[0]);
322  temp2 = (index_share[0] - additive_mask[0 + num_relu]);
323  temp3 = (0 - additive_mask[0 + num_relu]);
324  }
325  if (*xor_share == zero_small) {
326  ot_messages_0[0] = sci::makeBlock128(0ULL + temp0, 0ULL + temp2);
327  ot_messages_1[0] = sci::makeBlock128(0ULL + temp1, 0ULL + temp3);
328  } else {
329  ot_messages_0[0] = sci::makeBlock128(0ULL + temp1, 0ULL + temp3);
330  ot_messages_1[0] = sci::makeBlock128(0ULL + temp0, 0ULL + temp2);
331  }
332  }

334  void argmax_this_level_sub_32(type *argmax, type *result, type *indexshare,
335  type *share, int num_relu) {
336  uint8_t *drelu_ans = new uint8_t[num_relu];
337 
338  if (this->algeb_str == FIELD) {
339  relu_field_oracle->relu(result, share, num_relu, drelu_ans, true);
340  } else { // RING
341  relu_oracle->relu(result, share, num_relu, drelu_ans, true);
342  }
343 
344  // Now perform x.msb(x)
345  // 2 OTs required with reversed roles
346  uint64_t **ot_messages = new uint64_t *[num_relu];
347  for (int i = 0; i < num_relu; i++) {
348  ot_messages[i] = new uint64_t[2];
349  }
350  uint64_t *additive_masks = new uint64_t[2 * num_relu];
351 
352  uint64_t *received_shares = new uint64_t[num_relu];
353  uint64_t *received_shares_0 = new uint64_t[num_relu];
354  uint64_t *received_shares_1 = new uint64_t[num_relu];
355 
356  if (this->algeb_str == FIELD) {
357  this->relu_field_oracle->triple_gen->prg->template random_mod_p<type>(
358  (type *)additive_masks, 2 * num_relu, this->prime_mod);
359  for (int i = 0; i < 2 * num_relu; i++) {
360  additive_masks[i] %= this->prime_mod;
361  }
362  } else { // RING
363  this->relu_oracle->triple_gen->prg->random_data(
364  additive_masks, 2 * num_relu * sizeof(type));
365  }
366  if (party == sci::ALICE) {
367  for (int i = 0; i < num_relu; i++) {
368  set_argmax_end_ot_messages_sub_32(
369  ot_messages[i], share + i, indexshare + i, drelu_ans + i,
370  ((type *)additive_masks) + i, num_relu);
371  }
372  if (this->algeb_str == FIELD) {
373  relu_field_oracle->otpack->iknp_straight->send(ot_messages, num_relu,
374  this->l * 2);
375  relu_field_oracle->otpack->iknp_reversed->recv(
376  received_shares, drelu_ans, num_relu, this->l * 2);
377  } else { // RING
378  relu_oracle->otpack->iknp_straight->send(ot_messages, num_relu, 64);
379  relu_oracle->otpack->iknp_reversed->recv(received_shares, drelu_ans,
380  num_relu, 64);
381  }
382  } else // party = sci::BOB
383  {
384  for (int i = 0; i < num_relu; i++) {
385  set_argmax_end_ot_messages_sub_32(
386  ot_messages[i], share + i, indexshare + i, drelu_ans + i,
387  ((type *)additive_masks) + i, num_relu);
388  }
389  if (this->algeb_str == FIELD) {
390  relu_field_oracle->otpack->iknp_straight->recv(
391  received_shares, drelu_ans, num_relu, this->l * 2);
392  relu_field_oracle->otpack->iknp_reversed->send(ot_messages, num_relu,
393  this->l * 2);
394  } else { // RING
395  relu_oracle->otpack->iknp_straight->recv(received_shares, drelu_ans,
396  num_relu, 64);
397  relu_oracle->otpack->iknp_reversed->send(ot_messages, num_relu, 64);
398  }
399  }
400 
401  for (int i = 0; i < num_relu; i++) {
402  if (this->algeb_str == FIELD) {
403  // tightly optimized communication for the field case
404  received_shares_0[i] = (received_shares[i] & mask_upper) >> this->l;
405  received_shares_1[i] = received_shares[i] & mask_lower;
406  } else { // RING
407  received_shares_0[i] = (received_shares[i] & mask32_upper) >> 32;
408  received_shares_1[i] = received_shares[i] & mask32_lower;
409  }
410  }
411  for (int i = 0; i < num_relu; i++) {
412  result[i] = ((type *)additive_masks)[i] +
413  ((type *)received_shares_0)[(8 / sizeof(type)) * i];
414  argmax[i] = ((type *)additive_masks)[i + num_relu] +
415  ((type *)received_shares_1)[(8 / sizeof(type)) * i];
416  if (this->algeb_str == FIELD) {
417  result[i] %= this->prime_mod;
418  argmax[i] %= this->prime_mod;
419  }
420  }
421  delete[] additive_masks;
422  delete[] received_shares;
423  delete[] received_shares_0;
424  delete[] received_shares_1;
425  delete[] drelu_ans;
426  for (int i = 0; i < num_relu; i++) {
427  delete[] ot_messages[i];
428  }
429  delete[] ot_messages;
430  }

432  void set_argmax_end_ot_messages_sub_32(uint64_t *ot_messages,
433  type *value_share, type *index_share,
434  uint8_t *xor_share,
435  type *additive_mask, int num_relu) {
436  uint64_t temp0, temp1, temp2, temp3;
437  uint64_t mask_upper_general, mask_lower_general;
438  if (this->algeb_str == FIELD) {
439  temp0 = sci::neg_mod((int64_t)value_share[0] - (int64_t)additive_mask[0],
440  this->prime_mod);
441  temp1 = sci::neg_mod((int64_t)0LL - (int64_t)additive_mask[0],
442  this->prime_mod);
443  temp2 = sci::neg_mod((int64_t)index_share[0] -
444  (int64_t)additive_mask[0 + num_relu],
445  this->prime_mod);
446  temp3 = sci::neg_mod((int64_t)0LL - (int64_t)additive_mask[0 + num_relu],
447  this->prime_mod);
448  temp0 = temp0 << this->l;
449  temp1 = temp1 << this->l;
450  mask_upper_general = mask_upper;
451  mask_lower_general = mask_lower;
452  } else { // RING
453  temp0 = (type)(value_share[0] - additive_mask[0]);
454  temp0 = temp0 << 32;
455  temp1 = (type)(0 - additive_mask[0]);
456  temp1 = temp1 << 32;
457  temp2 = (type)(index_share[0] - additive_mask[0 + num_relu]);
458  temp3 = (type)(0 - additive_mask[0 + num_relu]);
459  mask_upper_general = mask32_upper;
460  mask_lower_general = mask32_lower;
461  }
462  if (*xor_share == zero_small) {
463  ot_messages[0] = (mask_upper_general & (0ULL + temp0)) ^
464  (mask_lower_general & (0ULL + temp2));
465  ot_messages[1] = (mask_upper_general & (0ULL + temp1)) ^
466  (mask_lower_general & (0ULL + temp3));
467  } else {
468  ot_messages[0] = (mask_upper_general & (0ULL + temp1)) ^
469  (mask_lower_general & (0ULL + temp3));
470  ot_messages[1] = (mask_upper_general & (0ULL + temp0)) ^
471  (mask_lower_general & (0ULL + temp2));
472  }
473  }
