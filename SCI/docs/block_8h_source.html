187 __attribute__((target("sse4.1,sse2"))) inline bool
188 cmpBlock(const block128 *x, const block128 *y, int nblocks) {
189  const block128 *dest = nblocks + x;
190  for (; x != dest;) {
191  __m128i vcmp = _mm_xor_si128(*(x++), *(y++));
192  if (!_mm_testz_si128(vcmp, vcmp))
193  return false;
194  }
195  return true;
196 }

251 // Modified from
252 // https://mischasan.wordpress.com/2011/10/03/the-full-sse2-bit-matrix-transpose-routine/
253 // with inner most loops changed to _mm_set_epi8 and _mm_set_epi16
254 #define INP(x, y) inp[(x)*ncols / 8 + (y) / 8]
255 #define OUT(x, y) out[(y)*nrows / 8 + (x) / 8]
256 
257 __attribute__((target("sse2"))) inline void
258 sse_trans(uint8_t *out, uint8_t const *inp, uint64_t nrows, uint64_t ncols) {
259  uint64_t rr, cc;
260  int i, h;
261  union {
262  __m128i x;
263  uint8_t b[16];
264  } tmp;
265  __m128i vec;
266  assert(nrows % 8 == 0 && ncols % 8 == 0);
267 
268  // Do the main body in 16x8 blocks:
269  for (rr = 0; rr <= nrows - 16; rr += 16) {
270  for (cc = 0; cc < ncols; cc += 8) {
271  vec = _mm_set_epi8(INP(rr + 15, cc), INP(rr + 14, cc), INP(rr + 13, cc),
272  INP(rr + 12, cc), INP(rr + 11, cc), INP(rr + 10, cc),
273  INP(rr + 9, cc), INP(rr + 8, cc), INP(rr + 7, cc),
274  INP(rr + 6, cc), INP(rr + 5, cc), INP(rr + 4, cc),
275  INP(rr + 3, cc), INP(rr + 2, cc), INP(rr + 1, cc),
276  INP(rr + 0, cc));
277  for (i = 8; --i >= 0; vec = _mm_slli_epi64(vec, 1))
278  *(uint16_t *)&OUT(rr, cc + i) = _mm_movemask_epi8(vec);
279  }
280  }
281  if (rr == nrows)
282  return;
283 
284  // The remainder is a block128 of 8x(16n+8) bits (n may be 0).
285  // Do a PAIR of 8x8 blocks in each step:
286  if ((ncols % 8 == 0 && ncols % 16 != 0) ||
287  (nrows % 8 == 0 && nrows % 16 != 0)) {
288  // The fancy optimizations in the else-branch don't work if the above
289  // if-condition holds, so we use the simpler non-simd variant for that case.
290  for (cc = 0; cc <= ncols - 16; cc += 16) {
291  for (i = 0; i < 8; ++i) {
292  tmp.b[i] = h = *(uint16_t const *)&INP(rr + i, cc);
293  tmp.b[i + 8] = h >> 8;
294  }
295  for (i = 8; --i >= 0; tmp.x = _mm_slli_epi64(tmp.x, 1)) {
296  OUT(rr, cc + i) = h = _mm_movemask_epi8(tmp.x);
297  OUT(rr, cc + i + 8) = h >> 8;
298  }
299  }
300  } else {
301  for (cc = 0; cc <= ncols - 16; cc += 16) {
302  vec = _mm_set_epi16(*(uint16_t const *)&INP(rr + 7, cc),
303  *(uint16_t const *)&INP(rr + 6, cc),
304  *(uint16_t const *)&INP(rr + 5, cc),
305  *(uint16_t const *)&INP(rr + 4, cc),
306  *(uint16_t const *)&INP(rr + 3, cc),
307  *(uint16_t const *)&INP(rr + 2, cc),
308  *(uint16_t const *)&INP(rr + 1, cc),
309  *(uint16_t const *)&INP(rr + 0, cc));
310  for (i = 8; --i >= 0; vec = _mm_slli_epi64(vec, 1)) {
311  OUT(rr, cc + i) = h = _mm_movemask_epi8(vec);
312  OUT(rr, cc + i + 8) = h >> 8;
313  }
314  }
315  }
316  if (cc == ncols)
317  return;
318 
319  // Do the remaining 8x8 block128:
320  for (i = 0; i < 8; ++i)
321  tmp.b[i] = INP(rr + i, cc);
322  for (i = 8; --i >= 0; tmp.x = _mm_slli_epi64(tmp.x, 1))
323  OUT(rr, cc + i) = _mm_movemask_epi8(tmp.x);
324 }

371 __attribute__((target("sse2"))) inline block128 double_block(block128 bl) {
372  const __m128i mask = _mm_set_epi32(135, 1, 1, 1);
373  __m128i tmp = _mm_srai_epi32(bl, 31);
374  tmp = _mm_and_si128(tmp, mask);
375  tmp = _mm_shuffle_epi32(tmp, _MM_SHUFFLE(2, 1, 0, 3));
376  bl = _mm_slli_epi32(bl, 1);
377  return _mm_xor_si128(bl, tmp);
378 }
