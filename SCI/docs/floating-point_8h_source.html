44 // A container to hold an array of floating-point values
45 // If party is set as PUBLIC for a FPArray instance, then the underlying array is known publicly and we maintain the invariant that both parties will hold identical data in that instance.
46 // Else, the underlying array is secret-shared and the class instance will hold the party's share of the secret array. In this case, the party data member denotes which party this share belongs to.
47 // m_bits denotes the mantissa bits (after the binary point; e.g. 23 for 32-bit floats) and e_bits denotes the exponent bits
48 // For efficiency reasons, we maintain the sign-bit (s; BoolArray), exponent (e; FixArray with ell = e_bits + 2 and s = 0) and mantissa (m; FixArray with ell = m_bits + 1 and s = m_bits) in different secret-shared arrays, and additionally maintain a zero-bit (z; BoolArray)

119  // if the input floating-point value overflows/underflows the representation, set it as inf/0
120  void check_bounds() {
121  for (int i = 0; i < size; i++) {
122  if (z[i] || (int64_t(e[i]) - int64_t(e_min()) < 0)) {
123  // 0 representation
124  e[i] = 0;
125  m[i] = 0;
126  }
127  if (int64_t(e[i]) - int64_t(e_max()) > 0) {
128  // inf representation
129  e[i] = e_max() + 1;
130  m[i] = 0;
131  }
132  }
133  }

135  // modulo mask for m
136  uint64_t m_mask() const {
137  return (1ULL << (this->m_bits + 1)) - 1;
138  }

140  // modulo mask for e
141  uint64_t e_mask() const {
142  return (uint64_t(1) << (this->e_bits + 2)) - 1;
143  }

145  uint64_t e_bias() const { return (uint64_t(1) << (this->e_bits - 1)) - 1; }

147  uint64_t e_min() const { return 1; }

149  uint64_t e_max() const { return 2 * e_bias(); }
