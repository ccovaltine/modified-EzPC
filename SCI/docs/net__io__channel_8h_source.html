50 class NetIO : public IOChannel<NetIO> {
51 public:
52  bool is_server;
53  int mysocket = -1;
54  int consocket = -1;
55  FILE *stream = nullptr;
56  char *buffer = nullptr;
57  bool has_sent = false;
58  string addr;
59  int port;
60  uint64_t counter = 0;
61  uint64_t num_rounds = 0;
62  bool FBF_mode;
63  LastCall last_call = LastCall::None;
64  NetIO(const char *address, int port, bool full_buffer = false,
65  bool quiet = false) {
66  this->port = port;
67  is_server = (address == nullptr);
68  if (address == nullptr) {
69  struct sockaddr_in dest;
70  struct sockaddr_in serv;
71  socklen_t socksize = sizeof(struct sockaddr_in);
72  memset(&serv, 0, sizeof(serv));
73  serv.sin_family = AF_INET;
74  serv.sin_addr.s_addr =
75  htonl(INADDR_ANY); /* set our address to any interface */
76  serv.sin_port = htons(port); /* set the server port number */
77  mysocket = socket(AF_INET, SOCK_STREAM, 0);
78  int reuse = 1;
79  setsockopt(mysocket, SOL_SOCKET, SO_REUSEADDR, (const char *)&reuse,
80  sizeof(reuse));
81  if (::bind(mysocket, (struct sockaddr *)&serv, sizeof(struct sockaddr)) <
82  0) {
83  perror("error: bind");
84  exit(1);
85  }
86  if (listen(mysocket, 1) < 0) {
87  perror("error: listen");
88  exit(1);
89  }
90  consocket = accept(mysocket, (struct sockaddr *)&dest, &socksize);
91  close(mysocket);
92  } else {
93  addr = string(address);
94 
95  struct sockaddr_in dest;
96  memset(&dest, 0, sizeof(dest));
97  dest.sin_family = AF_INET;
98  dest.sin_addr.s_addr = inet_addr(address);
99  dest.sin_port = htons(port);
100 
101  while (1) {
102  consocket = socket(AF_INET, SOCK_STREAM, 0);
103 
104  if (connect(consocket, (struct sockaddr *)&dest,
105  sizeof(struct sockaddr)) == 0) {
106  break;
107  }
108 
109  close(consocket);
110  usleep(1000);
111  }
112  }
113  set_nodelay();
114  stream = fdopen(consocket, "wb+");
115  buffer = new char[NETWORK_BUFFER_SIZE];
116  memset(buffer, 0, NETWORK_BUFFER_SIZE);
117  if (full_buffer) {
118  setvbuf(stream, buffer, _IOFBF, NETWORK_BUFFER_SIZE);
119  } else {
120  setvbuf(stream, buffer, _IONBF, NETWORK_BUFFER_SIZE);
121  }
122  this->FBF_mode = full_buffer;
123  if (!quiet)
124  std::cout << "connected\n";
125  }

127  void sync() {
128  int tmp = 0;
129  if (is_server) {
130  send_data(&tmp, 1);
131  recv_data(&tmp, 1);
132  } else {
133  recv_data(&tmp, 1);
134  send_data(&tmp, 1);
135  flush();
136  }
137  }

165  void flush() { fflush(stream); }

167  void send_data(const void *data, int len) {
168  if (last_call != LastCall::Send) {
169  num_rounds++;
170  last_call = LastCall::Send;
171  }
172  counter += len;
173  int sent = 0;
174  while (sent < len) {
175  int res = fwrite(sent + (char *)data, 1, len - sent, stream);
176  if (res >= 0)
177  sent += res;
178  else
179  fprintf(stderr, "error: net_send_data %d\n", res);
180  }
181  has_sent = true;
182  }
183 
184  void recv_data(void *data, int len) {
185  if (last_call != LastCall::Recv) {
186  num_rounds++;
187  last_call = LastCall::Recv;
188  }
189  if (has_sent)
190  fflush(stream);
191  has_sent = false;
192  int sent = 0;
193  while (sent < len) {
194  int res = fread(sent + (char *)data, 1, len - sent, stream);
195  if (res >= 0)
196  sent += res;
197  else
198  fprintf(stderr, "error: net_send_data %d\n", res);
199  }
200  }
