37 // A container to hold an array of fixed-point values
38 // If party is set as PUBLIC for a FixArray instance, then the underlying array is known publicly and we maintain the invariant that both parties will hold identical data in that instance.
39 // Else, the underlying array is secret-shared and the class instance will hold the party's share of the secret array. In this case, the party data member denotes which party this share belongs to.
40 // signed_ denotes the signedness, ell is the bitlength, and s is the scale of the underlying fixed-point array
41 // If s is set to 0, the FixArray will behave like an IntegerArray
42 class FixArray {
43 public:
44  int party = sci::PUBLIC;
45  int size = 0; // size of array
46  uint64_t *data = nullptr; // data (ell-bit integers)
47  bool signed_; // signed? (1: signed; 0: unsigned)
48  int ell; // bitlength
49  int s; // scale
50 
51  FixArray(){};
52 
53  FixArray(int party_, int sz, bool signed__, int ell_, int s_ = 0) {
54  assert(party_ == sci::PUBLIC || party_ == sci::ALICE || party_ == sci::BOB);
55  assert(sz > 0);
56  assert(ell_ <= 64 && ell_ > 0);
57  this->party = party_;
58  this->size = sz;
59  this->signed_ = signed__;
60  this->ell = ell_;
61  this->s = s_;
62  data = new uint64_t[sz];
63  }
