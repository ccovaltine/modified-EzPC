80 void funcReconstruct2PCCons(signedIntType *y, const intType *x, int len) {
81  intType temp = 0;
82  signedIntType ans = 0;
83  if (party == SERVER) {
84  io->send_data(x, len * sizeof(intType));
85  } else if (party == CLIENT) {
86  io->recv_data(y, len * sizeof(intType));
87  for (int i = 0; i < len; i++) {
88  temp = y[i] + x[i];
89  if (!isNativeRing) {
90  temp = sci::neg_mod(temp,
91  (int64_t)prime_mod); // cast temp to signed int and
92  // then take a proper modulo p
93  y[i] = getAnyRingSignedVal(temp);
94  } else {
95  temp = temp & (prime_mod - 1);
96  if (temp >= (prime_mod / 2))
97  y[i] = temp - prime_mod;
98  else
99  y[i] = temp;
100  }
101  }
102  }
103  return;
104 }

147 void funcTruncationIdeal(int size, intType *arr, int consSF) {
148  if (party == SERVER) {
149  io->send_data(arr, sizeof(intType) * size);
150  for (int i = 0; i < size; i++) {
151  arr[i] = 0;
152  }
153  } else {
154  intType *arrOther = new intType[size];
155  io->recv_data(arrOther, sizeof(intType) * size);
156  for (int i = 0; i < size; i++) {
157  signedIntType ans;
158  intType temp = arr[i] + arrOther[i];
159  if (!isNativeRing) {
160  temp = sci::neg_mod(temp, (int64_t)prime_mod);
161  ans = getAnyRingSignedVal(temp);
162  ans = div_floor(ans, 1ULL << consSF);
163  } else {
164  uint64_t moduloMask = (1ULL << bitlength) - 1;
165  if (bitlength == 64)
166  moduloMask = -1;
167  temp = temp & moduloMask;
168  ans = ((signedIntType)temp) >> consSF;
169  }
170  arr[i] = ans;
171  }
172  }

218 void funcMatmulThread(int tid, int N, int s1, int s2, int s3, intType *A,
219  intType *B, intType *C, int partyWithAInAB_mul) {
220  assert(tid >= 0);
221  int bucket_size = std::ceil(s2 / (double)N);
222  int s2StartIdx = tid * bucket_size; // Inclusive
223  int s2EndIdx = std::min((tid + 1) * bucket_size, s2); // Exclusive
224 
225 
226  if(s2StartIdx == s2EndIdx || s2StartIdx > s2){
227  memset(C, 0, s1 * s3 * sizeof(intType));
228  return;
229  }
230 
231  intType *APtr = new intType[s1 * (s2EndIdx - s2StartIdx)];
232  for (int i = 0; i < s1; i++) {
233  for (int j = s2StartIdx; j < s2EndIdx; j++) {
234  Arr2DIdxRowM(APtr, s1, (s2EndIdx - s2StartIdx), i, (j - s2StartIdx)) =
235  Arr2DIdxRowM(A, s1, s2, i, j);
236  }
237  }
238  intType *BPtr = B + s2StartIdx * s3;
239  /*
240  Case 1: If Alice has A and Bob has B
241  Then for even threads, Bob (holding B) acts as receiver and Alice
242  (holding A) acts as sender For odd threads, Bob (holding B) acts as sender
243  and Alice (holding A) acts as receiver.
244 
245  Case 2: If Alice has B and Bob has A
246  Then for even threads, Bob (holding A) acts as receiver and Alice
247  (holding B) as sender For odd threads, Bob (holding A) acts as sender and
248  Alice (holding B) as receiver.
249 
250  Note that for even threads, Bob is always the
251  receiver and Alice is always the sender and for odd threads, Bob is always
252  the sender and Alice the receiver This makes sure that the matmul and
253  otinstances (indexed by tid) are always used by with the same
254  sender-receiver pair.
255  */
256 
257 #ifdef USE_LINEAR_UNIFORM
258  assert(s2EndIdx > s2StartIdx);
259  bool useBobAsSender = tid & 1;
260  if (partyWithAInAB_mul == sci::BOB)
261  useBobAsSender = !useBobAsSender;
262  if (useBobAsSender) {
263  // Odd tid, use Bob (holding B) as sender and Alice (holding A) as receiver
264  if (party == partyWithAInAB_mul) {
265  multUniformArr[tid]->funcOTReceiverInputA(s1, (s2EndIdx - s2StartIdx), s3,
266  APtr, C, otInstanceArr[tid]);
267  } else {
268  multUniformArr[tid]->funcOTSenderInputB(s1, (s2EndIdx - s2StartIdx), s3,
269  BPtr, C, otInstanceArr[tid]);
270  }
271  } else {
272  // Even tid, use Bob (holding B) as receiver and Alice (holding A) as sender
273  if (party == partyWithAInAB_mul) {
274  multUniformArr[tid]->funcOTSenderInputA(s1, (s2EndIdx - s2StartIdx), s3,
275  APtr, C, otInstanceArr[tid]);
276  } else {
277  multUniformArr[tid]->funcOTReceiverInputB(s1, (s2EndIdx - s2StartIdx), s3,
278  BPtr, C, otInstanceArr[tid]);
279  }
280  }
281 #else // USE_LINEAR_UNIFORM
282  MultMode mode;
283 #ifdef TRAINING
284  mode = MultMode::None;
285 #else
286  if (tid & 1) {
287  if (partyWithAInAB_mul == sci::ALICE)
288  mode = MultMode::Bob_has_A;
289  else
290  mode = MultMode::Bob_has_B;
291  } else {
292  if (partyWithAInAB_mul == sci::ALICE)
293  mode = MultMode::Alice_has_A;
294  else
295  mode = MultMode::Alice_has_B;
296  }
297 #endif
298  if (s2EndIdx > s2StartIdx) {
299  multArr[tid]->matmul_cross_terms(s1, (s2EndIdx - s2StartIdx), s3, APtr,
300  BPtr, C, bitlength, bitlength, bitlength,
301  true, mode);
302  } else {
303  memset(C, 0, s1 * s3 * sizeof(intType));
304  }
305 #endif // USE_LINEAR_UNIFORM
306  delete[] APtr;
307 }

341 void funcTruncateTwoPowerRing(
342  int curParty, sci::IOPack *curiopack, sci::OTPack *curotpack,
343  sci::IKNP<sci::NetIO> *curiknp, ReLUProtocol<intType> *curReluImpl,
344  sci::PRG128 *curPrgInstance, int size, intType *inp, intType *outp,
345  int consSF, uint8_t *msbShare, bool doCarryBitCalculation = true) {
346  static const int rightShiftForMsb = bitlength - 1;
347  uint64_t *carryBitCompArr;
348  uint8_t *carryBitCompAns;
349  uint64_t moduloMask = sci::all1Mask(bitlength);
350 
351  for (int i = 0; i < size; i++) {
352  assert(inp[i] <= moduloMask);
353  }
354 
355  if (doCarryBitCalculation) {
356  carryBitCompArr = new uint64_t[size];
357  carryBitCompAns = new uint8_t[size];
358  for (int i = 0; i < size; i++) {
359  carryBitCompArr[i] = (inp[i] & sci::all1Mask(consSF));
360  if (curParty == sci::BOB) {
361  carryBitCompArr[i] = (sci::all1Mask(consSF)) - carryBitCompArr[i];
362  }
363  }
364  MillionaireProtocol millionaire(curParty, curiopack, curotpack);
365  millionaire.compare(carryBitCompAns, carryBitCompArr, size, consSF);
366  }
367 
368  bool createdMsbSharesHere = false;
369  if (msbShare == nullptr) {
370  msbShare = new uint8_t[size];
371  curReluImpl->relu(nullptr, inp, size, msbShare, true);
372  createdMsbSharesHere = true;
373  }
374 
375  if (curParty == sci::ALICE) {
376  uint64_t **otMessages1oo4 = new uint64_t *[size];
377  intType *otMessages1oo2Data = new intType[size];
378  intType *otMessages1oo2Corr = new intType[size];
379  intType *localShare1oo4 = new intType[size];
380  curPrgInstance->random_data(localShare1oo4, sizeof(intType) * size);
381 
382  for (int i = 0; i < size; i++) {
383  otMessages1oo4[i] = new uint64_t[4];
384  uint8_t localShareMSB = ((uint8_t)(inp[i] >> rightShiftForMsb));
385  intType r = -localShare1oo4[i];
386  for (int j = 0; j < 4; j++) {
387  uint8_t b0 = j & 1;
388  uint8_t b1 =
389  (j >> 1) &
390  1; // b1,b0 is the bit representation of j = [msb(a)]_1, msb(a_1)
391  uint8_t temp = ((msbShare[i] + b1 + localShareMSB) & 1) &
392  ((msbShare[i] + b1 + b0) & 1);
393  intType curMsg = r;
394  if (temp & (localShareMSB == 0)) {
395  // msb(a_0)=0, msb(a_1)=0, msb(a)=1
396  curMsg -= 1;
397  } else if (temp & (localShareMSB == 1)) {
398  // msb(a_0)=1, msb(a_1)=1, msb(a)=0
399  curMsg += 1;
400  }
401  // In other cases, extra term is 0
402  otMessages1oo4[i][j] = (uint64_t)curMsg;
403  }
404  if (doCarryBitCalculation) {
405  otMessages1oo2Corr[i] = ((intType)carryBitCompAns[i]);
406  }
407  }
408 
409  curotpack->kkot[1]->send_impl(otMessages1oo4, size, bitlength);
410  if (doCarryBitCalculation) {
411  curiknp->send_cot_moduloAdd<intType>(otMessages1oo2Data,
412  otMessages1oo2Corr, size);
413  }
414 
415  for (int i = 0; i < size; i++) {
416  intType shareTerm1 = ((intType)((getAnyRingSignedVal(inp[i])) >> consSF));
417  intType shareTerm2 = (localShare1oo4[i] << (bitlength - consSF));
418  outp[i] = shareTerm1 + shareTerm2;
419  if (doCarryBitCalculation) {
420  intType curCarryShare = -otMessages1oo2Data[i];
421  outp[i] = outp[i] + ((intType)carryBitCompAns[i]) - 2 * curCarryShare;
422  }
423  outp[i] = outp[i] & moduloMask;
424  delete[] otMessages1oo4[i];
425  }
426 
427  delete[] otMessages1oo4;
428  delete[] otMessages1oo2Data;
429  delete[] otMessages1oo2Corr;
430  delete[] localShare1oo4;
431  } else {
432  uint64_t *otRecvMsg1oo4 = new uint64_t[size];
433  uint8_t *choiceBits = new uint8_t[size];
434  intType *otRecvMsg1oo2 = new intType[size];
435  uint8_t *choiceBits1oo2OT = new uint8_t[size];
436  for (int i = 0; i < size; i++) {
437  uint8_t localShareMSB = ((uint8_t)(inp[i] >> rightShiftForMsb));
438  choiceBits[i] = (msbShare[i] << 1) + localShareMSB;
439  if (doCarryBitCalculation) {
440  choiceBits1oo2OT[i] = carryBitCompAns[i];
441  }
442  }
443 
444  curotpack->kkot[1]->recv_impl(otRecvMsg1oo4, choiceBits, size, bitlength);
445  if (doCarryBitCalculation) {
446  curiknp->recv_cot_moduloAdd<intType>(otRecvMsg1oo2, choiceBits1oo2OT,
447  size);
448  }
449  for (int i = 0; i < size; i++) {
450  intType G1 = (intType)otRecvMsg1oo4[i];
451  intType shareTerm1 = (getAnyRingSignedVal(inp[i])) >> consSF;
452  intType shareTerm2 = (G1 << (bitlength - consSF));
453  outp[i] = shareTerm1 + shareTerm2;
454  if (doCarryBitCalculation) {
455  outp[i] =
456  outp[i] + ((intType)carryBitCompAns[i]) - 2 * otRecvMsg1oo2[i];
457  }
458  outp[i] = outp[i] & moduloMask;
459  }
460 
461  delete[] otRecvMsg1oo4;
462  delete[] choiceBits;
463  delete[] otRecvMsg1oo2;
464  delete[] choiceBits1oo2OT;
465  }
466 
467  if (doCarryBitCalculation) {
468  delete[] carryBitCompArr;
469  delete[] carryBitCompAns;
470  }
471 
472  if (createdMsbSharesHere) {
473  delete[] msbShare;
474  }
475 }

515  - Currently if bitsForA + bitlength > 64, then the function resorts to

521 void funcAvgPoolTwoPowerRing(int curParty, sci::IOPack *curiopack,
522  sci::OTPack *curotpack,
523  sci::IKNP<sci::NetIO> *curiknp,
524  sci::KKOT<sci::NetIO> *curkkot,
525  ReLUProtocol<intType> *curReluImpl,
526  sci::PRG128 *curPrgInstance, int size,
527  intType *inp, intType *outp, intType divisor) {
528  assert(inp != outp &&
529  "Assumption is there is a separate array for input and output");
530  assert(divisor > 0 && "working with positive divisor");
531  static const int rightShiftForMsb = bitlength - 1;
532  assert(sci::all1Mask(bitlength) >
533  (6 * divisor - 1)); // 2^l-1 > 6d-1 => 2^l > 6d
534  intType ringRem, ringQuot;
535  if (bitlength == 64) {
536  ringRem = ((intType)(-divisor)) % divisor; //(2^l-d)%d = 2^l%d
537  ringQuot = (((intType)(-divisor)) / divisor) + 1; //((2^l-d)/d)+1 = 2^l/d
538  } else {
539  ringRem = prime_mod % divisor;
540  ringQuot = prime_mod / divisor;
541  }
542  uint8_t *msbShare = new uint8_t[size];
543  uint64_t moduloMask = sci::all1Mask(bitlength);
544 
545  for (int i = 0; i < size; i++) {
546  inp[i] = inp[i] & moduloMask;
547  intType shareTerm1 = div_floor(getAnyRingSignedVal(inp[i]), divisor);
548  intType localShareMSB = (inp[i] >> rightShiftForMsb);
549  intType shareTerm2 = div_floor(
550  (signedIntType)((inp[i] % divisor) - localShareMSB * ringRem), divisor);
551  outp[i] = shareTerm1 - shareTerm2;
552  if (curParty == sci::ALICE)
553  outp[i] += 1;
554  }
555 
556  curReluImpl->relu(nullptr, inp, size, msbShare, true);
557 
558  const uint64_t bitsForA = std::ceil(std::log2(6 * divisor));
559  uint64_t totalRandomBytesForLargeRing = sci::ceil_val(bitlength * size, 8);
560  uint8_t *localShareCorrPacked = new uint8_t[totalRandomBytesForLargeRing];
561  intType *localShareCorr = new intType[size];
562  int totalRandomBytesForSmallRing = sci::ceil_val(bitsForA * size, 8);
563  uint8_t *localShareCorrSmallRingPacked =
564  new uint8_t[totalRandomBytesForSmallRing];
565  uint64_t *localShareCorrSmallRing = new uint64_t[size];
566  intType *otMsgCorrRing = new intType[4 * size];
567  intType *otMsgCorrSmallRing = new intType[4 * size];
568 
569  bool OT1oo4FitsIn64Bits = ((bitsForA + bitlength) <= 64);
