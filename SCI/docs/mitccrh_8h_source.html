40 template <int BatchSize = 8> class MITCCRH {
41 public:
42  AES_KEY scheduled_key[BatchSize];
43  block128 keys[BatchSize];
44  int key_used = BatchSize;
45  block128 start_point;
46  uint64_t gid = 0;
47 
48  void setS(block128 sin) { this->start_point = sin; }
49 
50  void renew_ks(uint64_t gid) {
51  this->gid = gid;
52  renew_ks();
53  }
54 
55  void renew_ks() {
56  for (int i = 0; i < BatchSize; ++i)
57  keys[i] = start_point ^ makeBlock128(gid++, 0);
58  AES_opt_key_schedule<BatchSize>(keys, scheduled_key);
59  key_used = 0;
60  }
61 
62  template <int K, int H> void hash_cir(block128 *blks) {
63  for (int i = 0; i < K * H; ++i)
64  blks[i] = sigma(blks[i]);
65  hash<K, H>(blks);
66  }
67 
68  template <int K, int H> void hash(block128 *blks) {
69  assert(K <= BatchSize);
70  assert(BatchSize % K == 0);
71  if (key_used == BatchSize)
72  renew_ks();
73 
74  block128 tmp[K * H];
75  for (int i = 0; i < K * H; ++i)
76  tmp[i] = blks[i];
77 
78  ParaEnc<K, H>(tmp, scheduled_key + key_used);
79  key_used += K;
80 
81  for (int i = 0; i < K * H; ++i)
82  blks[i] = blks[i] ^ tmp[i];
83  }
84 };
