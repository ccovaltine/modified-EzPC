33 static inline int64_t signed_val_cleartext(uint64_t x, int bw_x) {
34  uint64_t pow_x = (bw_x == 64 ? 0ULL : (1ULL << bw_x));
35  uint64_t mask_x = pow_x - 1;
36  x = x & mask_x;
37  return int64_t(x - ((x >= (pow_x / 2)) * pow_x));
38 }

40 static int64_t lookup_sqrt(int32_t index, int m, int exp_parity) {
41  int32_t k = 1 << m;
42  double u = (1.0 + (double(index) / double(k))) * (1 << exp_parity);
43  double Y = 1.0 / sqrt(u);
44  int32_t scale = m + SQRT_LOOKUP_SCALE;
45  int64_t val = (Y * (1ULL << scale));
46  return val;
47 }

49 // A0 \in (1/4, 1)
50 static int64_t lookup_div_A0(int64_t index, int m) {
51  int32_t k = 1 << m;
52  double p = 1 + (double(index) / double(k));
53  double A1 = 1.0 / (p * (p + 1.0 / double(k)));
54  int32_t scale = m + 3;
55  int64_t val = (A1 * (1LL << scale));
56  return val;
57 }

59 // A1 \in (1/2, 1)
60 static int64_t lookup_div_A1(int64_t index, int m) {
61  int32_t k = 1 << m;
62  double p = 1 + (double(index) / double(k));
63  double z = (p * (p + (1.0 / double(k))));
64  double A1 = ((1.0 / double(k * 2)) + sqrt(z)) / z;
65  int32_t scale = 2 * m + 2;
66  int64_t val = (A1 * (1LL << scale));
67  return val;
68 }

70 static int64_t lookup_neg_exp(int64_t val_in, int32_t s_in, int32_t s_out) {
71  if (s_in < 0) {
72  s_in *= -1;
73  val_in *= (1 << (s_in));
74  s_in = 0;
75  }
76  int64_t res_val = exp(-1.0 * (val_in / double(1LL << s_in))) * (1LL << s_out);
77  return res_val;
78 }
